Linux x86_64 asm (nasm, intel-style)


			section .data
				msg db "Hello, World!",0ax
				msglen db $-msg

			section .text
				global _start
			_start:
				mov rax, 1 ; sys_write, See linux/arch/x86/entry/syscalls/syscalls_64.tbl
				mov rdi, 1 ; man write
				mov rsi, msg
				mov rdx, msglen
				syscall
				mov rax, 60 ; exit
				mov rdi, 0
				syscall

Linux x86_64 asm (gas)


				.globl _start
			.text
			_start:
				mov $1, %rax
				mov $1, %rdi
				mov $msg, %rsi
				mov $msglen, %rdx
				syscall

				mov $60, %rax
				mov $0, %rdi
				syscall
			.data
			msg:
				.ascii "Hello, World!\n"
				msglen = . - msg
